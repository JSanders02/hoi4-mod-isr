# Arab revolt - start
country_event = {
	id = israel.7
	title = israel.7.t
	desc = israel.7.d
	picture = GFX_report_event_spr_anarchist_uprising
	
	trigger = {
		tag = ISR
		AND = {
			OR = {
				has_full_control_of_state = 797
				has_full_control_of_state = 798
				PAL = {
					is_puppet_of = ISR
				}
			}
			OR = {
				has_full_control_of_state = 553
				has_full_control_of_state = 554
			}
		}
	}	
	
	fire_only_once = yes
	
	mean_time_to_happen = {
		days = 1000
		modifier = {
			factor = 0.001
			NOT = {
				country_exists = IRQ
			}
		}
	}
	
	option = {
		name = israel.7.a
		ISR = {
			remove_ideas = ISR_arabs_angry
			add_ideas = ISR_arabs_suppressed
		}
	}
	
	option = {
		name = israel.7.b
		custom_effect_tooltip = ISR_revolt_warning
		set_country_flag = ISR_revolt_ignored
	}
}
